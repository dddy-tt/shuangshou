- [ ] **Step 2: 写 `soft_uart.c`**

```c
#include "soft_uart.h"
#include "stm32f4xx_hal.h"
#include "stdio.h"
#include "string.h"

/* 9600bps 下每位 104.17μs
 * 系统滴答 HAL_GetTick 精度 1ms → 不够用
 * 用 DWT 周期计数器 (168MHz, 1 cycle = 5.95ns)
 * 104.17μs = 17500 cycles @ 168MHz */

#define BIT_CYCLES  17500U  /* 168MHz × 104.17μs */

static void delay_bit(void) {
    uint32_t start = DWT->CYCCNT;
    while ((DWT->CYCCNT - start) < BIT_CYCLES);
}

void SoftUART_Init(void) {
    /* PE0 初始化为推挽输出高电平 */
    GPIO_InitTypeDef g = {0};
    __HAL_RCC_GPIOE_CLK_ENABLE();
    g.Pin   = SOFTUART_TX_PIN;
    g.Mode  = GPIO_MODE_OUTPUT_PP;
    g.Pull  = GPIO_PULLUP;
    g.Speed = GPIO_SPEED_FREQ_MEDIUM;
    HAL_GPIO_Init(SOFTUART_TX_PORT, &g);
    SOFTUART_TX_PORT->BSRR = SOFTUART_TX_PIN;  /* 空闲高电平 */

    /* 使能 DWT 周期计数器 */
    CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
    DWT->CYCCNT = 0;
    DWT->CTRL  |= DWT_CTRL_CYCCNTENA_Msk;
}

void SoftUART_SendByte(uint8_t byte) {
    /* 起始位: 拉低 */
    SOFTUART_TX_PORT->BSRR = (uint32_t)SOFTUART_TX_PIN << 16U;
    delay_bit();
    /* 8 数据位, LSB first */
    for (uint8_t i = 0; i < 8; i++) {
        if (byte & 0x01)
            SOFTUART_TX_PORT->BSRR = SOFTUART_TX_PIN;
        else
            SOFTUART_TX_PORT->BSRR = (uint32_t)SOFTUART_TX_PIN << 16U;
        delay_bit();
        byte >>= 1;
    }
    /* 停止位: 拉高 */
    SOFTUART_TX_PORT->BSRR = SOFTUART_TX_PIN;
    delay_bit();
}

void SoftUART_SendBuf(const uint8_t *buf, uint16_t len) {
    for (uint16_t i = 0; i < len; i++)
        SoftUART_SendByte(buf[i]);
}

/* 构造 MQTT PUBLISH 帧 (QoS 0, 无 retain) */
void SoftUART_SendMQTT(const char *topic, const char *payload) {
    uint8_t frame[128];
    uint16_t pos = 0;
    uint16_t tlen = strlen(topic);
    uint16_t plen = strlen(payload);

    /* MQTT PUBLISH 固定头 */
    frame[pos++] = 0x30;  /* PUBLISH, QoS0, DUP=0, Retain=0 */

    /* 剩余长度 = 2+topic + 2+payload */
    uint8_t rlen = (uint8_t)(2 + tlen + plen);
    frame[pos++] = rlen;

    /* Topic 长度 + Topic */
    frame[pos++] = (uint8_t)(tlen >> 8);
    frame[pos++] = (uint8_t)(tlen & 0xFF);
    memcpy(&frame[pos], topic, tlen); pos += tlen;

    /* Payload */
    memcpy(&frame[pos], payload, plen); pos += plen;

    SoftUART_SendBuf(frame, pos);
}
```

function parseFrame(data) {
  const v=new Uint8Array(data.buffer);
  if(v.byteLength<18||v[0]!==0xAA||v[17]!==0xBB) return;
  let csum=0;for(let i=1;i<17;i++)csum^=v[i];
  if(csum!==v[16])return;
  for(let i=0;i<5;i++)document.getElementById('rf'+i).style.height=(v[1+i]/100*100)+'%';
  for(let i=0;i<5;i++)document.getElementById('lf'+i).style.height=(v[13+i]/100*100)+'%';
  let roll=(v[6]|(v[7]<<8))/100, pitch=(v[8]|(v[9]<<8))/100, yaw=(v[10]|(v[11]<<8))/100;
  if(roll>180)roll-=655.36;if(pitch>180)pitch-=655.36;if(yaw>180)yaw-=655.36;
  drawAttitude(roll,pitch,yaw);
  document.getElementById('att-text').textContent=`R:${roll.toFixed(1)}° P:${pitch.toFixed(1)}° Y:${yaw.toFixed(1)}°`;
  document.getElementById('hr-val').textContent=v[12];
  document.getElementById('spo2-val').textContent=v[13];
  const modes=['translate','control','rehab'],names=['翻译模式','控制模式','康复模式'];
  document.getElementById('mode-badge').className='mode-badge '+modes[v[14]];
  document.getElementById('mode-badge').textContent=names[v[14]];
}

async function connect(){
  try{
    document.getElementById('status').textContent='🟡 搜索中...';
    dev=await navigator.bluetooth.requestDevice({filters:[{services:[BLE_SERVICE]}]});
    const srv=await dev.gatt.connect();
    const svc=await srv.getPrimaryService(BLE_SERVICE);
    char=await svc.getCharacteristic(BLE_CHAR);
    await char.startNotifications();
    char.addEventListener('characteristicvaluechanged',e=>parseFrame(e.target.value));
    document.getElementById('status').textContent='🟢 已连接';
  }catch(e){
    document.getElementById('status').textContent='🔴 未连接 - 点我重试';
  }
}

document.getElementById('status').onclick=connect;
connect();
